007package io.pivotal.rti.protocols;
008
009import java.util.Map;
010
011/**
012 * @author Soby Chacko
013 */
014public final class ProtocolEventDefaults {
015
016    private static final double LATITUDE = 90;
017    private static final double LONGITUDE = 180;
018    private static final double RADII90 = -1;
019
020    private static final Integer FIRST_LAC = 0;
021    private static final Integer FIRST_CELL_ID = 0;
022    private static final Long DEFAULT_STATUS = 0L;
023    private static final int[] DEFAULT_EXT_TAGS = {};
024
025    private ProtocolEventDefaults() {
026    }
027
028    public static void populateDetailMapWithMissingDefaultValues(Map<String, Object> protocolDetails) {
029        addDefaultLatitude(protocolDetails);
030        addDefaultLongitude(protocolDetails);
031        addDefaultRadii90(protocolDetails);
032        addDefaultFirstLac(protocolDetails);
033        addDefaultFirstCellId(protocolDetails);
034        addDefaultLac(protocolDetails);
035        addDefaultCellId(protocolDetails);
036        addDefaultStatus(protocolDetails);
037        addDefaultExtTags(protocolDetails);
038    }
039
040    static void addDefaultLatitude(Map<String,Object> protocolDetails) {
041        if(protocolDetails.get("latitude") == null){
042            protocolDetails.put("latitude", LATITUDE);
043        }
044    }
045
046    static void addDefaultLongitude(Map<String,Object> protocolDetails) {
047        if(protocolDetails.get("longitude") == null){
048            protocolDetails.put("longitude", LONGITUDE);
049        }
050    }
051
052    static void addDefaultRadii90(Map<String,Object> protocolDetails) {
053        if(protocolDetails.get("radii90") == null){
054            protocolDetails.put("radii90", RADII90);
055        }
056    }
057
058    static void addDefaultFirstLac(Map<String,Object> protocolDetails) {
059        if(protocolDetails.get("firstLac") == null){
060            protocolDetails.put("firstLac", FIRST_LAC);
061        }
062    }
063
064    static void addDefaultFirstCellId(Map<String,Object> protocolDetails) {
065        if(protocolDetails.get("firstCellId") == null){
066            protocolDetails.put("firstCellId", FIRST_CELL_ID);
067        }
068    }
069
070    static void addDefaultLac(Map<String,Object> protocolDetails) {
071        if(protocolDetails.get("lac") == null){
072            protocolDetails.put("lac", FIRST_LAC);
073        }
074    }
075
076    static void addDefaultCellId(Map<String,Object> protocolDetails) {
077        if(protocolDetails.get("cellId") == null){
078            protocolDetails.put("cellId", FIRST_CELL_ID);
079        }
080    }
081
082    static void addDefaultStatus(Map<String,Object> protocolDetails) {
083        if(protocolDetails.get("status") == null){
084            protocolDetails.put("status", DEFAULT_STATUS);
085        }
086    }
087
088    static void addDefaultExtTags(Map<String,Object> protocolDetails) {
089        if(protocolDetails.get("extTags") == null){
090            protocolDetails.put("extTags", DEFAULT_EXT_TAGS);
091        }
092    }
093
094}
