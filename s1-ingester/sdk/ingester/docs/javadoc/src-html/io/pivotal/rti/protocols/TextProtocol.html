009import java.lang.annotation.Documented;
010import java.lang.annotation.ElementType;
011import java.lang.annotation.Inherited;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016/**
017 * In order to implement a text based protocol adapter,
018 * the {@link ProtocolAdapter}
019 * must be annotated with {@link TextProtocol}. This is necessary because the
020 * Ingester server will then switch over to a line based channel instead
021 * of the frame based channels used for binary protocols (which is the default).
022 *
023 * @author Soby Chacko
024 */
025@Target(ElementType.TYPE)
026@Retention(RetentionPolicy.RUNTIME)
027@Inherited
028@Documented
029public @interface TextProtocol {
030}
