007package io.pivotal.rti.protocols;
008
009import java.lang.annotation.Documented;
010import java.lang.annotation.ElementType;
011import java.lang.annotation.Retention;
012import java.lang.annotation.RetentionPolicy;
013import java.lang.annotation.Target;
014
015/**
016 * If the ProtocolAdapter implementation has constructor parameters and if the Spring auto-configuration mechanism
017 * is <b>not</b> used, then you must tag your constructor with {@link ConstructorArgumentKeys} annotation
018 * and provide keys that correspond to the parameters in the declaration order. These keys must correspond
019 * to the same keys in the properties file that is provided in the resources directory.
020 *
021 * Note: The usage of this annotation is completely optional and only required if you are not using Spring
022 * and having constructor with arguments.
023 *
024 * @author Soby Chacko
025 */
026@Documented
027@Retention(RetentionPolicy.RUNTIME)
028@Target(ElementType.CONSTRUCTOR)
029public @interface ConstructorArgumentKeys {
030    String[] keys();
031}
