011/**
012 * Core domain class provided through the codecs API and needed for implementing
013 * a valid {@link ProtocolAdapter}.
014 *
015 * ProtocolEvent captures domain agnostic data from a network protocol. For
016 * example, it gives an abstraction over telecommunication protocols or any other protocols
017 * from a different domain.
018 *
019 * ProtocolEvent has two layers to it. First, it has a few native fields those
020 * are common to many different kind of protocols. For example, it has a primaryId, serviceId,
021 * sourceId, temporaryId, ingestTime, other timing attributes, initial/current location
022 * id's from where the protocol occurred etc. Please see the individual methods for
023 * further explanation. Second, it has a {@link ProtocolDetailMap} that contains a key/value
024 * pair of protocol specific information. This is where the implementers of new
025 * Codecs have to put data that cannot natively be placed in a ProtocolEvent.
026 *
027 * @author Charlie Black
028 * @author manoharanv
029 * @author Martin Sustrik
030 * @author Soby Chacko
031 *
032 */
033public class ProtocolEvent {
034
035    private String key;
036    private String primaryId = null;
037    private String serviceId = null;
038    private String temporaryId;
039    private String sourceId = null;
040    private long ingestTime = System.currentTimeMillis();
041    private long eventStartTime = 0;
042    private long eventEndTime = 0;
043    private String initialLocationId = "N/A";
044    private String currentLocationId = "N/A";
045    private String protocolName = "N/A";
046    private short protocolType = 0;
047    private short tag = 0;
048    private String serviceTarget = null;
049
050    private ProtocolAttributeMap protocolAttributeMap = new ProtocolAttributeMap();
051    private ProtocolDetailMap protocolDetails = new ProtocolDetailMap();
052
053    /**
054     * Creates a basic empty ProtocolEvent.
055     */
056    public ProtocolEvent() {
057        super();
058        ProtocolEventDefaults.populateDetailMapWithMissingDefaultValues(protocolDetails);
059    }

061    /**

075     * Codec through the adapter implementation.

078     *            Name of the protocol

080    public void setProtocolName(String protocolName) {
081        this.protocolName = protocolName;
082    }
083
084    /**
085     *
086     * @return key of this event
087     */
088    public final String getKey() {
089        return key;
090    }
091
092    /**
093     * Key makes a protocol event unique from the rest. So care must be taken
094     * when
095     * generating this. Possible keys can be, but not limited to are,
096     * UUID's, randomly generated strings etc. The implementer has to make sure
097     * the
098     * uniqueness of the key.
099     *
100     * @param key
101     *            unique key for this ProtocolEvent.
102     */
103    public final void setKey(String key) {
104        this.key = key;
105    }

108     *
109     * @return the time the event was ingested to RTI
110     */

123        this.ingestTime = ingestTime;

126    /**

128     * @return service identifier for this ProtocolEvent

130    public final String getServiceId() {

133
134    /**

144     */

147    }
148
149    /**
150     *
151     * @return primary identifier for this event
152     */
153    public final String getPrimaryId() {

155    }
156
157    /**
158     * primaryId is, as the name indicates, is the primary identifier for this
159     * ProtocolEvent.
160     * In telco domains, this corresponds to IMSI (International Mobile
161     * Subscriber Identity)

163     *
164     * @param primaryId
165     *            primary identifier for this event
166     */
167    public final void setPrimaryId(String primaryId) {
168        this.primaryId = primaryId;
169    }
170

172     *
173     * @return temporary identifier for this event

181     * ProtocolEvent.

185     * @param temporaryId
186     *            temporaryId for this event
187     */
188    public final void setTemporaryId(String temporaryId) {
189        this.temporaryId = temporaryId;
190    }

193     *
194     * @return source identifier of this event

197        return sourceId;
198    }

201     * sourceId is from where this event is originated from. In telco domains,

205     * @param sourceId
206     *            source Id for this event
207     */
208    public final void setSourceId(String sourceId) {
209        this.sourceId = sourceId;
210    }

212    /**
213     *

215     */

217        return serviceTarget;

220    /**
221     * serviceTarget refers to the entity that this ProtocolEvent is targeting

224     *

227     */
228    public final void setServiceTarget(String serviceTarget) {

383                ", eventStartTime=" + getEventStartTime() +

391    @Override
392    public boolean equals(Object o) { //NOSONAR
393        if (this == o) {
394            return true;
395        }
396        if (o == null || getClass() != o.getClass()) {
397            return false;
398        }
399
400        ProtocolEvent that = (ProtocolEvent) o;
401
402        if (key != null ? !key.equals(that.key) : that.key != null) {
403            return false;
404        }
405        if (primaryId != null ? !primaryId.equals(that.primaryId) : that.primaryId != null) {
406            return false;
407        }
408        if (sourceId != null ? !sourceId.equals(that.sourceId) : that.sourceId != null) {
409            return false;
410        }
411
412        if (eventStartTime != that.eventStartTime) {
413            return false;
414        }
415        if (eventEndTime != that.eventEndTime) {
416            return false;
417        }
418        if (ingestTime != that.ingestTime) {
419            return false;
420        }
421        if (protocolType != that.protocolType) {
422            return false;
423        }
424
425        if (protocolName != null ? !protocolName.equals(that.protocolName) : that.protocolName != null) {
426            return false;
427        }
428        if (serviceId != null ? !serviceId.equals(that.serviceId) : that.serviceId != null) {
429            return false;
430        }
431        if (temporaryId != null ? !temporaryId.equals(that.temporaryId) : that.temporaryId != null) {
432            return false;
433        }
434
435        if (serviceTarget != null ? !serviceTarget.equals(that.serviceTarget) : that.serviceTarget != null) {
436            return false;
437        }
438        if (currentLocationId != null ? !currentLocationId.equals(that.currentLocationId) : that.currentLocationId != null) {
439            return false;
440        }
441        if (initialLocationId != null ? !initialLocationId.equals(that.initialLocationId) : that.initialLocationId != null) {
442            return false;
443        }
444        if (protocolDetails != null ? !protocolDetails.equals(that.protocolDetails) : that.protocolDetails != null) {
445            return false;
446        }
447        if (protocolAttributeMap != null ? !protocolAttributeMap.equals(that.protocolAttributeMap) : that.protocolAttributeMap != null) {
448            return false;
449        }
450        return true;
451    }
452
453    @Override
454    public int hashCode() { //NOSONAR
455        int result;
456        result = key != null ? key.hashCode() : 0;
457        result = 31 * result + (serviceId != null ? serviceId.hashCode() : 0);
458        result = 31 * result + (primaryId != null ? primaryId.hashCode() : 0);
459        result = 31 * result + (temporaryId != null ? temporaryId.hashCode() : 0);
460        result = 31 * result + (sourceId != null ? sourceId.hashCode() : 0);
461        result = 31 * result + (int) (ingestTime ^ (ingestTime >>> 32));
462        result = 31 * result + (int) (eventStartTime ^ (eventStartTime >>> 32));
463        result = 31 * result + (int) (eventEndTime ^ (eventEndTime >>> 32));
464        result = 31 * result + (initialLocationId != null ? initialLocationId.hashCode() : 0);
465        result = 31 * result + (currentLocationId != null ? currentLocationId.hashCode() : 0);
466        result = 31 * result + (protocolName != null ? protocolName.hashCode() : 0);
467        result = 31 * result + (int) protocolType;
468        result = 31 * result + (protocolAttributeMap != null ? protocolAttributeMap.hashCode() : 0);
469        result = 31 * result + (protocolDetails != null ? protocolDetails.hashCode() : 0);
470        result = 31 * result + (serviceTarget != null ? serviceTarget.hashCode() : 0);
471        return result;
472    }
